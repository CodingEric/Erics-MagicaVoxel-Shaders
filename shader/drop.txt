bool chk(vec3 v){
    if(voxel(vec3(v.x,v.y,v.z+1.))==iArgs[0]&&voxel(v)==0.){
        return true;
    }
    return false;
}

float map(vec3 v){
    if(voxel(v)==iArgs[0]){
        return 0.;
    }else if(chk(v)){
        return iArgs[0];
    }else{
        return voxel(v);
    }
}
